\begin{frame}[fragile]
\frametitle{Helper Function}

Helper function to parse the mail

\begin{lstlisting}[language={Python},basicstyle={\footnotesize}]
def parse_mail(text):
    zeilen = text.split('\n') # create array
    daten = {} # empty dict

    for zeile in zeilen:
        splits = zeile.split(':')
        if len(splits) == 2:
            daten[splits[0].strip()] = splits[1].strip()
            
    return daten
\end{lstlisting}
